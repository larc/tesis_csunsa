\def\images#1#2#3#4
{
	\begin{figure}[!htb]
		\centering
		\foreach \imagepath/\imagename/\imagecaption in #2
		{
			\subcaptionbox{\label{\imagepath/\imagename} \em \imagecaption}
			{\includegraphics[width=#1\textwidth]{\imagepath/\imagename}}
		}
		\caption{\em #4}\label{#3}
	\end{figure}
}
